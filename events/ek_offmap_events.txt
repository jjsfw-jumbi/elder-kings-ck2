#Sacrifice a character
character_event = {
	id = ekardor.20
	has_dlc = "Jade Dragon"
    desc = EKArdor20
	picture = GFX_evt_two_moon_aurora
	
	is_triggered_only = yes
	hide_from = yes
	show_from_from = yes
	
	option = {
		name = EKArdor20Accept
		death = {
			death_reason = death_sacrificed
			killer = event_target:oblivion_sacrifice_killer
		}
		mother = {
			opinion = {
				modifier = opinion_killed_close_kin
				who = event_target:oblivion_sacrifice_killer
			}
		}
		father = {
			opinion = {
				modifier = opinion_killed_close_kin
				who = event_target:oblivion_sacrifice_killer
			}
		}
		spouse = {
			opinion = {
				modifier = opinion_killed_close_kin
				who = event_target:oblivion_sacrifice_killer
			}
		}
		any_child = {
			opinion = {
				modifier = opinion_killed_close_kin
				who = event_target:oblivion_sacrifice_killer
			}
		}
		if = {
			limit = {
				NOT = { dynasty = event_target:oblivion_sacrifice_killer }
				NOT = { sibling = event_target:oblivion_sacrifice_killer }
			}
			any_sibling = {
				opinion = {
					modifier = opinion_killed_close_kin
					who = event_target:oblivion_sacrifice_killer
				}
			}
			any_dynasty_member = {
				opinion = {
					modifier = opinion_mutilated_family
					who = event_target:oblivion_sacrifice_killer
				}
			}
		}
	}
}

# MASS SACRIFICE
character_event = {
	id = ekardor.30
	has_dlc = "Jade Dragon"
    desc = EKArdor30
	picture = GFX_evt_two_moon_aurora
	
	is_triggered_only = yes
	show_from_from = no
	
	immediate = {
		set_variable = { which = local_mass_sacrifice_result value = 0 }
		
		while = {
			limit = {
				any_courtier = {
					prisoner = yes
					is_adult = yes
					NOT = { is_heir = yes }
					NOT = { has_character_flag = he_ded }
					NOT = { has_character_flag = saved_from_mass_sacrifice }
					
					# Can't be a summonded entity, obviously
					NOR = {
						has_opinion_modifier = { who = ROOT modifier = opinion_summoner }
						has_opinion_modifier = { who = ROOT modifier = opinion_bound_dremora }
					}
					
					# Can't have already been taken into account
					NOT = { has_character_flag = mass_sacrifice_counted }
				}
			}
		
			random_courtier = {
				limit = {
					prisoner = yes
					is_adult = yes
					NOT = { is_heir = yes }
					NOT = { has_character_flag = he_ded }
					NOT = { has_character_flag = saved_from_mass_sacrifice }
					
					# Can't be a summonded entity, obviously
					NOR = {
						has_opinion_modifier = { who = ROOT modifier = opinion_summoner }
						has_opinion_modifier = { who = ROOT modifier = opinion_bound_dremora }
					}
					
					# Can't have already been taken into account
					NOT = { has_character_flag = mass_sacrifice_counted }
				}
				
				### Compute how much ardor that would give
				set_variable = { which = local_mass_sacrifice_local value = 25 }
				## Stats - dip
				if = {
					limit = { diplomacy = 8 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { diplomacy = 12 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { diplomacy = 16 }
					change_variable = { which = local_mass_sacrifice_local value = 10 }
				}
				
				if = {
					limit = { diplomacy = 20 }
					change_variable = { which = local_mass_sacrifice_local value = 30 }
				}
				## Stats - mar
				if = {
					limit = { martial = 8 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { martial = 12 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { martial = 16 }
					change_variable = { which = local_mass_sacrifice_local value = 10 }
				}
				
				if = {
					limit = { martial = 20 }
					change_variable = { which = local_mass_sacrifice_local value = 30 }
				}
				## Stats - ste
				if = {
					limit = { stewardship = 8 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { stewardship = 12 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { stewardship = 16 }
					change_variable = { which = local_mass_sacrifice_local value = 10 }
				}
				
				if = {
					limit = { stewardship = 20 }
					change_variable = { which = local_mass_sacrifice_local value = 30 }
				}
				## Stats - int
				if = {
					limit = { intrigue = 8 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { intrigue = 12 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { intrigue = 16 }
					change_variable = { which = local_mass_sacrifice_local value = 10 }
				}
				
				if = {
					limit = { intrigue = 20 }
					change_variable = { which = local_mass_sacrifice_local value = 30 }
				}
				## Stats - lea
				if = {
					limit = { learning = 8 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { learning = 12 }
					change_variable = { which = local_mass_sacrifice_local value = 5 }
				}
				
				if = {
					limit = { learning = 16 }
					change_variable = { which = local_mass_sacrifice_local value = 10 }
				}
				
				if = {
					limit = { learning = 20 }
					change_variable = { which = local_mass_sacrifice_local value = 30 }
				}
				## Tier
				if = {
					limit = { tier == BARON }
					change_variable = { which = local_mass_sacrifice_local value = 25 }
				}
				if = {
					limit = { tier == COUNT }
					change_variable = { which = local_mass_sacrifice_local value = 75 }
				}
				if = {
					limit = { tier == DUKE }
					change_variable = { which = local_mass_sacrifice_local value = 150 }
				}
				if = {
					limit = { tier == KING }
					change_variable = { which = local_mass_sacrifice_local value = 400 }
				}
				if = {
					limit = { tier == EMPEROR }
					change_variable = { which = local_mass_sacrifice_local value = 1000 }
				}
				## Dragonblood
				if = {
					limit = { trait = dragonborn }
					change_variable = { which = local_mass_sacrifice_local value = 2500 }
				}
				if = {
					limit = { trait = dragonborn_descendant }
					change_variable = { which = local_mass_sacrifice_local value = 250 }
				}
				## Bloodline
				if = {
					limit = { has_bloodline = yes }
					change_variable = { which = local_mass_sacrifice_local value = 100 }
				}
				
				# We apply the local result
				change_variable = { which = local_mass_sacrifice_result which = local_mass_sacrifice_local }
				
				set_character_flag = mass_sacrifice_counted
			}
		}
		
		ROOT = { save_event_target_as = oblivion_sacrifice_killer }
	}
	
	option = {
		name = EKArdor30Accept
		
		sound_effect = china_grace_gain
		
		if = {
			limit = { NOT = { has_character_flag = brotherhood_qualified } }
			custom_tooltip = {
				text = brotherhood_qualification_notification
				hidden_tooltip = {
					set_character_flag = brotherhood_qualified
				}
			}
		}
		
		clr_character_flag = oblivion_mass_sacrifice_ongoing
		
		any_courtier = {
			limit = { has_character_flag = mass_sacrifice_counted }
			set_character_flag = he_ded
			character_event = { id = ekardor.20 }
		}
		
		add_offmap_currency = {
			offmap = FROM
			value = local_mass_sacrifice_result
		}
		
		if = {
			limit = {
				FROM = { is_offmap_tag = oblivion_ideal_masters }
			}
			oblivion_ideal_masters = {
				offmap_ruler = {
					change_variable = { which = offmap_souls which = local_mass_sacrifice_result }
					change_variable = { which = souls_from_sacrifice which = local_mass_sacrifice_result }
				}
			}
		}
	}
}
