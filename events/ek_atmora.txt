# Fires on startup
province_event = {
        id = atmora.0
        hide_window = yes
        is_triggered_only = yes

        only_playable = yes

        trigger = {
                is_save_game = no
                province_id = 1
        }

        immediate = {
                any_province = {
                        limit = {
                                region = nirn_atmora
                        }
                        add_province_modifier = {
				name = freezing
				duration = -1
			}
                }
        }
}

# Atmoran expedition: Select leader of an expedition
narrative_event = {
	id = atmora.2
        title = atmora.2.title
	desc = atmora.2
	picture = GFX_evt_trade_post_republic
        is_triggered_only = yes
        hide_new = yes

        # Select random couriter
        immediate = {
                hidden_tooltip = {
                        random_courtier = {
                                limit = {
                                        is_adult = yes
                                        religion = ROOT
                                        NOR = {
                                                trait = incapable
                                                trait = craven
                                                is_inaccessible_trigger = yes
                                        }
                                        prisoner = no
                                        max_age = 50
                                        min_age = 20
                                }
                                preferred_limit = {
                                        stewardship = 5
                                        OR = {
                                               trait = ravager
                                               trait = viking
                                               trait = seaking
                                               trait = sea_queen
                                               trait = pirate
                                        }
                                }
                                preferred_limit = {
                                        stewardship = 7
                                        OR = {
                                               trait = trader_1
                                               trait = trader_2
                                               trait = trader_3
                                               trait = adventurer
                                               trait = scout_1
                                               trait = scout_2
                                               trait = scout_3
                                        }
                                }
                                preferred_limit = {
                                        stewardship = 7
                                        combat_rating = 10
                                }
                                save_event_target_as = expedition_leader_courtier
                        }

                        create_character = {
                                random_traits = yes
                                dynasty = actually_culture
                                culture = atmoran
                                religion = animisim
                                female = no
                                min_age = 25
                                max_age = 50
                        }
                        new_character = {
                                save_event_target_as = expedition_leader_random
                        }
                }
        }

        # Select one of your courtiers
        option = {
                name = atmora.2.opta
                trigger = {
                        event_target:expedition_leader_courtier = { always = yes }
                }
                event_target:expedition_leader_courtier = {
                        tooltip = { add_trait = on_adventure }
                        save_event_target_as = atmoran_expedition_leader
                }
                # Kill the new character
                hidden_tooltip = {
                        event_target:expedition_leader_random = {
                                death = { death_reason = death_missing }
                        }
                }
        }

        # Select a voulenteer from the hold
        option = {
                name = atmora.2.optb
                event_target:expedition_leader_random = {
                        tooltip = { add_trait = on_adventure }
                        save_event_target_as = atmoran_expedition_leader
                }
        }

        # Select the outcome and report it to ruler
        after = {
                hidden_tooltip = { 
                        event_target:atmoran_expedition_leader = {
                                narrative_event = { id = atmora.3 }
                        }
                }
                clear_event_target = expedition_leader_courtier
                clear_event_target = expedition_leader_random
                hidden_tooltip = { narrative_event = { id = atmora.4 days = 120 random = 90 }}
        }
}

# For atmoran provinces with freezing modifier
#
# Depopulation due to freeezing climate
# What if people switch 
province_event = {
        id = atmora.7
        picture = GFX_evt_undead2
        desc = atmora.7

        trigger = {
                region = nirn_atmora
                NAND = {
                        has_global_flag = atmora_reconnected 
                        OR = {
                                port = yes
                                has_trade_post = yes
                                any_neighbor_province = {
                                        has_trade_post = yes
                                }
                        }
                }
                NOT = { has_province_modifier = depopulated_3 }
                NOT = { has_province_modifier = depopulated_province }
                # Undead provinces do not get depopulated
                NOT = { culture_group = undead }
                NOT = { culture_group = daedra }
        }

        mean_time_to_happen = {
                years = 80
                modifier = {
                        factor = 0.5
                        port = no
                }
                # Extra time with no prosperity buffer
                modifier = {
                        factor = 3.0
                        NOT = { has_dlc = "Reapers" }
                }
                modifier = {
                        factor = 0.8
                        holder_scope = {
                                NOT = { stewardship = 5 }
                        }
                }
                modifier = {
                        factor = 1.5
                        is_capital = yes
                }
                modifier = {
                        factor = 1.5
                        OR = { 
                                culture = atmoran
                                culture = skaal
                                culture = nord
                        }
                }
                modifier = {
                        factor = 1.2
                        holder_scope = {
                                stewardship = 10
                        }
                }
                modifier = {
                        factor = 1.2
                        holder_scope = {
                                stewardship = 15
                        }
                }
        }

        option = {
                name = atmora.7.opta
                atmoran_depopulation_effect = yes       
        }
}
