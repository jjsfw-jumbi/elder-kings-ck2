####This is a general event used for all public societies such as the Mages Guild and Companions
character_event = {
    id = eksocieties.10
	hide_window = yes
    is_triggered_only = yes
    trigger = {
		OR = {
			FROM = { leader = { society_member_of = fighters_guild_tamriel } }
			FROM = { leader = { society_member_of = fighters_guild_akavir } }
			# FROM = { leader = { society_member_of = fighters_guild_atmora } }
			FROM = { leader = { society_member_of = companions_tamriel } }
			FROM = { leader = { society_member_of = ansei_guild_tamriel } }
			FROM = { leader = { society_member_of = mages_guild_tamriel } }
			FROM = { leader = { society_member_of = mages_guild_akavir } }
			FROM = { leader = { society_member_of = mercantile_guild_tamriel } }
			FROM = { leader = { society_member_of = mercantile_guild_akavir } }
			FROM = { leader = { society_member_of = scenarist_guild_tamriel } }
			FROM = { leader = { society_member_of = scenarist_guild_akavir } }
			FROM = { leader = { society_member_of = moth_priests_tamriel } }
			FROM = { leader = { society_member_of = greybeards_tamriel } }
			FROM = { leader = { society_member_of = house_dres_council } }
			FROM = { leader = { society_member_of = house_hlaalu_council } }
			FROM = { leader = { society_member_of = house_indoril_council } }
			FROM = { leader = { society_member_of = house_redoran_council } }
			FROM = { leader = { society_member_of = house_telvanni_council } }
			FROM = { leader = { society_member_of = house_sadras_council } }
			FROM = { leader = { society_member_of = house_sul_council } }
			FROM = { leader = { society_member_of = house_veloth_council } }
			FROM = { leader = { society_member_of = house_dagoth_council } }
			FROM = { leader = { society_member_of = elder_council } }
		}
    }
    immediate = {
		if = {
			limit = { FROM = { leader = { society_member_of = greybeards_tamriel } } }
			set_character_flag = greybeards_initiation
		}
		if = {
			limit = { FROM = { leader = { society_member_of = elder_council } } }
			if = {
				limit = {
					OR = {
						has_global_flag = ec_policy_join_vote_emp
						has_global_flag = ec_policy_join_appoint_emp
					}
					NOT = { # If you can apply to the HC, apply to them. They'll generally be much more likely to accept
						has_global_flag = ec_policy_join_vote_hc
						has_global_flag = ec_policy_join_appoint_hc
					}
				}
				ec_get_emperor_target = yes
				event_target:target_emperor = {
					character_event = { id = eldercouncil.1 }
				}
			}
			else = {
				FROM = { leader = { character_event = { id = eldercouncil.1 } } }
			}
			break = yes
		}
    	FROM = { leader = { character_event = { id = eksocieties.11 } } }
    }
}

### Exchanging society currency for tech points
# MAGES ARE EXCLUDED. They're OP enough as it is.
# Moth Priests can get tech points through reading the Elder Scrolls, so again, not included
# Agent guilds (Assassins & Thieves) get a random outcome among:
	# Random tech points
	# Random tech boost in the capital
character_event = {
	id = eksocieties.250
	desc = eksocieties.250.d
	is_triggered_only = yes
	
	picture = GFX_evt_library
	
	# Buy military tech points, full
	option = {
		trigger = {
			OR = {
				is_member_of_fighters_guild = yes
				society_member_of = companions_tamriel
			}
		}
		
		name = eksocieties.250.mil
		military_techpoints = 50
	}
	
	# Buy economic tech points, full
	option = {
		trigger = {
			OR = {
				is_member_of_mercantile_guild = yes
				society_member_of = elder_council
			}
		}
		
		name = eksocieties.250.eco
		economy_techpoints = 50
	}
	
	# Buy cultural tech points, full
	option = {
		trigger = {
			OR = {
				is_member_of_scenarist_guild = yes
				society_member_of = greybeards_tamriel
			}
		}
		
		name = eksocieties.250.cul
		culture_techpoints = 50
	}
	
	# Buy military & cultural, full
	option = {
		trigger = {
			society_member_of = ansei_guild_tamriel
		}
		
		name = eksocieties.250.milcul
		military_techpoints = 25
		culture_techpoints = 25
	}
	
	# Agent bordering a province with better tech
	option = {
		trigger = {
			OR = {
				is_member_of_assassins_guild = yes
				is_member_of_thieves_guild = yes
			}
			
			any_realm_province = {
				any_neighbor_province = {
					has_higher_tech_than = ROOT
					NOT = { has_province_modifier = sabotage_research }
				}
			}
		}
		
		name = eksocieties.250.agentBetterTech
		random_list = {
			33 = {
				military_techpoints = 50
			}
			33 = {
				economy_techpoints = 50
			}
			33 = {
				culture_techpoints = 50
			}
		}
		
		random_realm_province = {
			limit = {
				any_neighbor_province = {
					has_higher_tech_than = ROOT
				}
			}
			random_neighbor_province = {
				limit = {
					has_higher_tech_than = ROOT
					NOT = { has_province_modifier = sabotage_research }
				}
				
				add_province_modifier = {
					name = sabotage_research
					duration = 1800
				}
			}
		}
	}
	
	# Agent not bordering a province with better tech
	option = {
		trigger = {
			OR = {
				is_member_of_assassins_guild = yes
				is_member_of_thieves_guild = yes
			}
		}
		
		name = eksocieties.250.agent
		random_list = {
			33 = {
				military_techpoints = 30
			}
			33 = {
				economy_techpoints = 30
			}
			33 = {
				culture_techpoints = 30
			}
		}
	}
	
	# Nevermind
	option = {
		name = eksocieties.250.cancel
		
		clr_character_flag = exchanging_influence_for_tech
	}
	
	after = {
		# Apply the modifier
		hidden_tooltip = {
			if = {
				limit = { has_character_flag = exchanging_influence_for_tech }
				add_character_modifier = {
					name = society_advancements_purchased
					duration = 3650
				}
				detract_society_currency_major_effect_scaling = yes
				clr_character_flag = exchanging_influence_for_tech
			}
		}
	}
}

### Strengthen body
# Feeble -> Nothing -> Sturdy -> Robust
character_event = {
	id = eksocieties.260
	picture = GFX_evt_warrior
	
	is_triggered_only = yes
	
	desc = {
		text = eksocieties.260.feeble
		trigger = { trait = feeble }
	}
	
	desc = {
		text = eksocieties.260.sturdy
		trigger = { trait = sturdy }
	}
	
	desc = {
		text = eksocieties.260.nothing
		trigger = {
			NOR = {
				trait = feeble
				trait = sturdy
			}
		}
	}
	
	option = {
		name = eksocieties.260.optA
		
		if = {
			limit = { trait = feeble }
			remove_trait = feeble
		}
		
		else_if = {
			limit = { trait = sturdy }
			add_trait = robust
		}
		
		else = {
			add_trait = sturdy
		}
	}
}
